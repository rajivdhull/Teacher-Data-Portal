// Max file size via portal (must match MAX_UPLOAD_MB in Code.gs)

const PORTAL_MAX_MB = 30;


function fAccept(f){const m={image:'image/*',pdf:'application/pdf',doc:'.doc,.docx,.xls,.xlsx,.ppt,.pptx','image,pdf':'image/*,application/pdf','image,doc':'image/*,.doc,.docx,.xls,.xlsx,.ppt,.pptx',any:'*'};return m[f.allowedTypes||'any']||'*';}

function fSel(inp,fid,maxF,maxM){processF(Array.from(inp.files),fid,maxF,maxM);inp.value='';}

function fDrop(e,fid,maxF,maxM,accept){processF(Array.from(e.dataTransfer.files),fid,maxF,maxM);}

function processF(files,fid,maxF,maxM){

  if(!uploads[fid])uploads[fid]=[];

  // Effective max: lower of field setting and portal hard limit

  const effMax=Math.min(maxM,PORTAL_MAX_MB);

  for(const f of files){

    if(uploads[fid].length>=maxF){showE(`Max ${maxF} files allowed.`);break;}

    const fileMB=f.size/1024/1024;

    if(fileMB>effMax){

      // For large videos, offer the Google Drive link fallback

      const isVideo=f.type.startsWith('video/');

      if(isVideo){

        showLargeFileHelper(fid,f.name,fileMB,effMax);

      }else{

        showE(`"${f.name}" is ${fileMB.toFixed(1)}MB — exceeds the ${effMax}MB limit. Please compress the file and try again.`);

      }

      continue;

    }

    uploads[fid].push(f);

  }

  rPrev(fid);

}

async function b64(file){return new Promise((res,rej)=>{const r=new FileReader();r.onload=()=>res(r.result.split(',')[1]);r.onerror=rej;r.readAsDataURL(file);});}


async function upFiles(fid,label){

  const files=uploads[fid]||[];

  const links=(driveLinks[fid]||[]).slice(); // already-collected external links

  const pr=document.getElementById('fpr_'+fid);if(pr)pr.style.display='block';


  for(let i=0;i<files.length;i++){

    const f=files[i];

    if(pr)pr.textContent=`Uploading ${i+1}/${files.length}: ${f.name}…`;

    try{

      const d=await b64(f);

      const r=await apiPost({action:'uploadFile',fileName:f.name,mimeType:f.type,data:d,folder:label});

      if(r.link)links.push(r.link);

      else if(r.viewLink)links.push(r.viewLink);

      else if(r.error)throw new Error(r.error);

      if(r.warning)toast(r.warning,'err');

    }catch(e){

      showE('Upload failed: '+e.message);

      if(pr)pr.style.display='none';

      return null;

    }

  }

  if(pr)pr.style.display='none';

  return links.join(', ');

}
